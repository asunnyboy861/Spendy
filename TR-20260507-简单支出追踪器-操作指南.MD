### 3.3 ViewModel核心逻辑

```swift
import SwiftData
import Foundation

@Observable
final class ExpenseViewModel {
    var todayTotal: Double = 0
    var weekTotal: Double = 0
    var monthTotal: Double = 0
    var recentExpenses: [Expense] = []

    private var modelContext: ModelContext?

    func configure(modelContext: ModelContext) {
        self.modelContext = modelContext
        fetchTotals()
    }

    func addExpense(amount: Double, category: ExpenseCategory,
                    note: String = "") {
        guard let context = modelContext else { return }
        let expense = Expense(
            amount: amount,
            category: category.rawValue,
            note: note
        )
        context.insert(expense)
        try? context.save()
        fetchTotals()
    }

    func deleteExpense(_ expense: Expense) {
        guard let context = modelContext else { return }
        context.delete(expense)
        try? context.save()
        fetchTotals()
    }

    func fetchTotals() {
        guard let context = modelContext else { return }
        let now = Date()
        let calendar = Calendar.current

        let todayStart = calendar.startOfDay(for: now)
        let weekStart = calendar.date(from: calendar.dateComponents(
            [.yearForWeekOfYear, .weekOfYear], from: now))!
        let monthStart = calendar.date(from: calendar.dateComponents(
            [.year, .month], from: now))!

        let descriptor = FetchDescriptor<Expense>(
            sortBy: [SortDescriptor(\.date, order: .reverse)]
        )
        let allExpenses = (try? context.fetch(descriptor)) ?? []

        todayTotal = allExpenses
            .filter { $0.date >= todayStart }
            .reduce(0) { $0 + $1.amount }

        weekTotal = allExpenses
            .filter { $0.date >= weekStart }
            .reduce(0) { $0 + $1.amount }

        monthTotal = allExpenses
            .filter { $0.date >= monthStart }
            .reduce(0) { $0 + $1.amount }

        recentExpenses = Array(allExpenses.prefix(5))
    }
}
```

```swift
import SwiftUI

struct QuickAddExpenseView: View {
    let viewModel: ExpenseViewModel
    @Binding var newAmount: String
    @Binding var selectedCategory: ExpenseCategory
    @Environment(\.dismiss) private var dismiss
    @FocusState private var isAmountFocused: Bool

    var body: some View {
        NavigationStack {
            VStack(spacing: 24) {

                    Text("$")
                        .font(.system(size: 40, weight: .bold))
                        .foregroundStyle(.secondary)

                        .font(.system(size: 48, weight: .bold))
                        .keyboardType(.decimalPad)

                .padding(.horizontal, 24)

                LazyVGrid(columns: Array(repeating: GridItem(.flexible()),
                                         count: 4), spacing: 12) {
                    ForEach(ExpenseCategory.allCases) { category in
                        CategoryButton(
                            category: category,
                            isSelected: selectedCategory == category
                        ) {
                            selectedCategory = category
                        }
                    }
                }
                .padding(.horizontal, 20)

                Button {
                    if let amount = Double(newAmount), amount > 0 {
                        viewModel.addExpense(
                            amount: amount,
                            category: selectedCategory
                        )
                        newAmount = ""
                        dismiss()
                    }
                } label: {
                    Text("Add")
                        .font(.headline)
                        .frame(maxWidth: .infinity)
                        .padding(.vertical, 16)
                        .background(Color.blue)
                        .foregroundStyle(.white)
                        .clipShape(RoundedRectangle(cornerRadius: 14))

                .disabled(Double(newAmount) == nil || Double(newAmount) == 0)
            }
            .navigationTitle("Add Expense")
            .navigationBarTitleDisplayMode(.inline)
            .onAppear { isAmountFocused = true }
        }
    }
```

### 3.6 Widget实现

```swift
import WidgetKit
import SwiftUI

struct SpendyWidget: Widget {
    let kind: String = "SpendyWidget"

    var body: some WidgetConfiguration {
        StaticConfiguration(kind: kind, provider: SpendyProvider()) { entry in
            SpendyWidgetEntryView(entry: entry)
        }
        .configurationDisplayName("Spendy")
        .description("See your spending at a glance.")
        .supportedFamilies([.systemSmall, .systemMedium])
    }
}

struct SpendyEntry: TimelineEntry {
    let date: Date
    let todayTotal: Double
    let weekTotal: Double
    let monthTotal: Double
}

struct SpendyProvider: TimelineProvider {
    func placeholder(in context: Context) -> SpendyEntry {
        SpendyEntry(date: .now, todayTotal: 0, weekTotal: 0, monthTotal: 0)
    }

    func getSnapshot(in context: Context,
                     completion: @escaping (SpendyEntry) -> Void) {
        let totals = fetchTotalsFromAppGroup()
        completion(SpendyEntry(
            date: .now,
            todayTotal: totals.today,
            weekTotal: totals.week,
            monthTotal: totals.month
        ))
    }

    func getTimeline(in context: Context,
                     completion: @escaping (Timeline<SpendyEntry>) -> Void) {
        let totals = fetchTotalsFromAppGroup()
        let entry = SpendyEntry(
            date: .now,
            todayTotal: totals.today,
            weekTotal: totals.week,
            monthTotal: totals.month
        )
        let nextUpdate = Calendar.current.date(
            byAdding: .hour, value: 1, to: .now
        )!
        completion(Timeline(entries: [entry], policy: .after(nextUpdate)))
    }

    private func fetchTotalsFromAppGroup() -> (today: Double,
                                                 week: Double,
                                                 month: Double) {
        guard let defaults = UserDefaults(
            suiteName: "group.com.spendy.app"
        ) else { return (0, 0, 0) }
        return (
            defaults.double(forKey: "todayTotal"),
            defaults.double(forKey: "weekTotal"),
            defaults.double(forKey: "monthTotal")
        )
    }
}

struct SpendyWidgetEntryView: View {
    let entry: SpendyEntry

    var body: some View {
        VStack(alignment: .leading, spacing: 8) {
            Text("Spendy")

                Label("Today", systemImage: "calendar")
```

### 3.7 编写规则

| 规则 | 说明 |
|------|------|
| 架构 | 严格MVVM：View只绑定ViewModel，ViewModel操作Model |
| 数据流 | 单向数据流：View → Action → ViewModel → Model → ViewModel更新 → View刷新 |
| 命名 | View以View结尾，ViewModel以ViewModel结尾，Model用@Model标注 |
| 错误处理 | 使用try?处理非关键错误，关键路径用do-catch+用户提示 |
| 无第三方依赖 | 零CocoaPods/SPM依赖，纯Apple原生框架 |
| 隐私 | 所有数据本地存储，App Group共享给Widget，不联网 |
| 无注释 | 代码自文档化，函数/变量名即文档 |

```
┌─────────────────────────────────────────────────────────────┐
│                    Spendy 数据流架构                          │
└─────────────────────────────────────────────────────────────┘

用户输入层                    逻辑层                     数据层
═══════════              ═══════════               ═══════════

┌──────────┐           ┌──────────────┐          ┌──────────────┐
│          │  Action   │              │  CRUD    │              │
│  Views   │──────────▶│  ViewModel   │─────────▶│  SwiftData   │
│          │           │              │          │  /CoreData   │
│ ContentView│          │ ExpenseVM    │          │              │
│ QuickAdd  │◀─────────│              │◀─────────│  Expense     │
│ WidgetView│  @Query   │              │  Fetch   │  Entity      │
│ Settings  │  Observer │              │  Result  │              │
└──────────┘           └──────┬───────┘          └──────┬───────┘
                              │                         │
                              │ UserDefaults            │ App Group
                              │ (Widget数据)            │ (共享数据)
                              ▼                         ▼
                       ┌──────────────┐          ┌──────────────┐
                       │  WidgetKit   │          │  App Group   │
                       │  Timeline    │          │  UserDefaults│
                       │  Provider    │◀─────────│  todayTotal  │
                       │              │          │  weekTotal   │
                       └──────────────┘          │  monthTotal  │
                                                 └──────────────┘

数据流详细路径:
━━━━━━━━━━━━━━

1. 添加支出:
   用户点击"Add"
   → QuickAddExpenseView 调用 viewModel.addExpense()
   → ViewModel 创建 Expense 对象 → context.insert()
   → SwiftData 持久化到本地SQLite
   → ViewModel.fetchTotals() 重新计算
   → 更新 todayTotal/weekTotal/monthTotal
   → 同步写入 App Group UserDefaults (供Widget读取)
   → View 自动刷新显示新数字
   → Widget 通过 Timeline 刷新显示

2. 删除支出:
   用户滑动删除
   → View 调用 viewModel.deleteExpense()
   → ViewModel 调用 context.delete()
   → SwiftData 删除记录
   → fetchTotals() 重新计算
   → 同步 App Group
   → View + Widget 刷新

3. Widget读取:
   Widget Timeline Provider 触发
   → 读取 App Group UserDefaults
   → 构建 SpendyEntry
   → 渲染 WidgetView
   → 每小时自动刷新 + App写入时触发

4. Siri快捷指令:
   用户语音 "log $15 food to Spendy"
   → AppIntent 处理
   → 通过 ModelContext 插入 Expense
   → 同步 App Group
   → Widget + View 刷新
```

## 七、极具竞争力的价格策略

### 7.1 定价策略选择: 完全免费（超级简单，用来引流）

| 维度 | 分析 |
|------|------|
| **项目定位** | 超级简单支出追踪器，核心功能=3个数字+快速输入 |
| **功能复杂度** | 极简，无API费用，无服务器成本，纯本地运行 |
| **开发成本** | 1-2周开发，零运营成本 |
| **竞争环境** | YNAB $99/年, Monarch $14/月, Copilot $8/月, Expenses $1.99/月 |

### 7.2 为什么选择"完全免费"而非其他模式

| 模式 | 适用场景 | Spendy适用性 | 结论 |
|------|---------|-------------|------|
| **完全免费** | 超级简单App，用来引流 | ✅ 完美匹配：功能极简+零运营成本+引流价值 | ✅ 采用 |
| 下载前付费 | 简单软件，付费后所有功能可用 | ❌ 用户对$0.99以上的未知App付费意愿低，无试用 | ❌ 不采用 |
| 站内订阅A | 免费下载+免费功能+一次性买断 | ❌ 功能太简单，无法拆分免费/付费，用户会觉得"就这还要钱？" | ❌ 不采用 |
| 站内订阅B | 复杂功能+月/年/买断 | ❌ 不符合极简定位 | ❌ 不采用 |

### 7.3 免费策略的商业逻辑
